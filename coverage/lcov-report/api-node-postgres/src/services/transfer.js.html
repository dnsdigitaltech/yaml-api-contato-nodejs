const ValidationError = require('../errors/ValidationError');
 
module.exports = (app) => {
  const find = (filter = {}) => {
    return app.db('transfers')
      .where(filter)
      .select();
  };
  const findOne = (filter = {}) => {
    return app.db('transfers')
      .where(filter)
      .first();
  };
 
  // metodo para validação
  const validate = async (transfer) => {
    // antes de salvar fazer as verificações
    // se não enviar a descrição então lance a mensagem
    if (!transfer.description) throw new ValidationError('Descrição é uma atributo obrigatório');
    if (!transfer.ammount) throw new ValidationError('Valor é uma atributo obrigatório');
    if (!transfer.date) throw new ValidationError('Data é uma atributo obrigatório');
    if (!transfer.acc_ori_id) throw new ValidationError('Conta de origem é uma atributo obrigatório');
    if (!transfer.acc_dest_id) throw new ValidationError('Conta de destino é uma atributo obrigatório');
    if (transfer.acc_ori_id == transfer.acc_dest_id) throw new ValidationError('Não é possível tranferir de uma conta para ele mesma');
 
    // consulta na conta
    const accounts = await app.db('accounts').whereIn('id', [transfer.acc_dest_id, transfer.acc_ori_id]);
    // busquei as duas contas agora farei um laço em cima delas
    accounts.forEach((acc) => {
      if (acc.user_id != parseInt(transfer.user_id, 10)) throw new ValidationError(`Conta de nº ${acc.id} não pertence ao usuário`);
    });
  };
 
  const save = async (transfer) => {
    // await validate(transfer); método no roteador
 
    const result = await app.db('transfers').insert(transfer, '*');
    const transferId = result[0].id;
    // transações que serão geradas por conta desta transferência
    const transactions = [
      { description: `Transfer to acc #${transfer.acc_dest_id}`, date: transfer.date, ammount: transfer.ammount * -1, type: 'O', acc_id: transfer.acc_ori_id, transfer_id: transferId, status: true },
      { description: `Transfer from acc #${transfer.acc_ori_id}`, date: transfer.date, ammount: transfer.ammount, type: 'I', acc_id: transfer.acc_dest_id, transfer_id: transferId, status: true },
    ];
    // inserir as transações
    await app.db('transactions').insert(transactions);
    return result;
  };
 
  const update = async (id, transfer) => {
    // await validate(transfer); método no roteador'
 
    const result = await app.db('transfers')
      .where({ id })
      .update(transfer, '*');
 
    const transactions = [
      { description: `Transfer to acc #${transfer.acc_dest_id}`, date: transfer.date, ammount: transfer.ammount * -1, type: 'O', acc_id: transfer.acc_ori_id, transfer_id: id, status: true },
      { description: `Transfer from acc #${transfer.acc_ori_id}`, date: transfer.date, ammount: transfer.ammount, type: 'I', acc_id: transfer.acc_dest_id, transfer_id: id, status: true },
    ];
    // antesn de inderir remover o que existe
    await app.db('transactions').where({ transfer_id: id }).del();// filtro nas transações que possuem ref, para esta transf
    await app.db('transactions').insert(transactions);
    return result;
  };
 
  const remove = async (id) => {
    // remover as transações
    await app.db('transactions').where({ transfer_id: id }).del();
    return app.db('transfers').where({ id }).del();
  };
  return { find, save, findOne, update, validate, remove };
};
